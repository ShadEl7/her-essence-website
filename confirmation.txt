    <h1>Thank You for Your Order!</h1>
    <p>Your payment was successful. Below are your order details:</p>

      <!-- Order details will be dynamically loaded here -->
    </div>

    <p>If you have any questions, please contact us at support@example.com.</p>
  </div>

  <script>
    // Retrieve the session ID from the URL
    const urlParams = new URLSearchParams(window.location.search);
    const sessionId = urlParams.get('session_id');

    // Fetch order details from the server using the session ID
    async function fetchOrderDetails() {
      try {
        const response = await fetch(`/order-details?session_id=${sessionId}`);
        const order = await response.json();

        const orderDetailsContainer = document.getElementById('orderDetails');
        orderDetailsContainer.innerHTML = `
          <p><strong>Order ID:</strong> ${order.id}</p>
          <p><strong>Total:</strong> ${order.amount_total / 100} ${order.currency.toUpperCase()}</p>
          <p><strong>Items:</strong></p>
          <ul>
            ${order.items.map(item => `<li>${item.quantity} x ${item.name}</li>`).join('')}
          </ul>
        `;
      } catch (error) {
        console.error('Error fetching order details:', error);
        alert('An error occurred while fetching your order details.');
      }
    }

    fetchOrderDetails();
  </script>
